```c++
inline int cc(int n,int m) {
	int sum = 1;
	int k = 1;
	if (m==0) {
		return 1;
	}
	for (int i = n;i>=n-m+1;--i) {
		sum = (sum * i) % mod;
	}
	for (int i = 1;i<=m;++i) {
		k = (k * i) % mod;
	}
	return (sum * inv(k, mod))%mod;
}
```
